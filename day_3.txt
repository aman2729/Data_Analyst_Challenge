
Q)Given Table Student_details,Write a query which displays names alternately by gender and sorted by ascending order of columnID

CREATE TABLE students (
    ID INT PRIMARY KEY,
    Name VARCHAR(50),
    Gender CHAR(1)
);

-- Insert the data into the student_details table
INSERT INTO students (ID, Name, Gender) VALUES

WITH ranked_table
AS
(
    SELECT *,
        ROW_NUMBER() OVER(PARTITION BY gender ORDER BY id) as rn
    FROM students
),
ranked_table2
AS
(
SELECT 
    id,
    name,
    gender,
    rn
FROM ranked_table    
ORDER BY rn,
         CASE
             WHEN gender='M' THEN 1 
             ELSE 2
         END   
)

SELECT
    id,
    name,
    gender
FROM ranked_table2
